Binary Search
(recursive approach)

Problem:

Find a particular value in a sorted array, return -1 if it is not in the array

Approach:

The easy approach towards this problem would be the linear approach, which is essentially a brute force approach. Where we can iterate through the array one by one, and touch every single element until we find a match, or if we hit the end of the array, and if we still haven't found a match, then it is not within the array and we just return false.

But that would take O(n) time, we can do better.

By if we implement Binary search, we can cut down the time from O(n) to O(logn)

Code Solution:
          
    var arr = [1,10,20,47,59,63,75,88,99,107];

    function binarySearch(arr, min, max, val) {
      if (min > max) return -1;
      let mid = min + Math.floor(max - min / 2);
      if (val > arr[mid]) {
        min = mid + 1;
        return binarySearch(arr, min, max, val);
      } else if (val < arr[mid]) {
        max = mid - 1;
        return binarySearch(arr, min, max, val);
      } else if (val === arr[mid]) {
        return arr[mid];
      }
    }
    var val = 47;
    var bSearch = binarySearch(arr, 0, arr.length - 1, val);
    if (bSearch !== -1) {
      console.log(`Value found! ${val} is in the array`);
    } else {
      console.log(`Value ${val} is not in the array`);
